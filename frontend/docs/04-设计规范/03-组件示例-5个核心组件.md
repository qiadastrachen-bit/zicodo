### Step1：创建/重写 `Button.vue`（按钮组件）

#### 1.1 组件功能
- 支持 3 种尺寸：`sm`（小）、`md`（中）、`lg`（大）
- 支持 3 种类型：`primary`（主要）、`secondary`（次要）、`ghost`（幽灵）
- 支持禁用状态：`disabled`
- 支持图标按钮：`icon`（圆形图标按钮）

#### 1.2 组件代码

**文件路径**：`src/components/Button.vue`

```vue
<template>
  <button
    :class="[
      'zl-button',
      `zl-button--${size}`,
      `zl-button--${type}`,
      { 'zl-button--disabled': disabled }
    ]"
    :disabled="disabled"
    @click="handleClick"
  >
    <slot name="icon-left"></slot>
    <span class="zl-button__text">
      <slot>按钮</slot>
    </span>
    <slot name="icon-right"></slot>
  </button>
</template>

<script setup>
/**
 * Button 按钮组件
 * @param {string} size - 尺寸：sm | md | lg
 * @param {string} type - 类型：primary | secondary | ghost
 * @param {boolean} disabled - 是否禁用
 */
const props = defineProps({
  size: {
    type: String,
    default: 'md',
    validator: (val) => ['sm', 'md', 'lg'].includes(val)
  },
  type: {
    type: String,
    default: 'primary',
    validator: (val) => ['primary', 'secondary', 'ghost'].includes(val)
  },
  disabled: {
    type: Boolean,
    default: false
  }
})

const emit = defineEmits(['click'])

function handleClick(event) {
  if (!props.disabled) {
    emit('click', event)
  }
}
</script>
```

### Step5：创建 `EmptyState.vue`（空状态组件）

#### 5.1 组件功能
- 显示固定素材 `loading.png` 或 `error.png`
- 支持自定义文案
- 支持轮播文案（参考 `page-prompt-team-page-revision.md`）

#### 5.2 组件代码

**文件路径**：`src/components/EmptyState.vue`

```vue
<template>
  <div class="empty-state">
    <img :src="imageSrc" :alt="altText" class="empty-state__image" />
    <p class="empty-state__text">{{ currentMessage }}</p>
  </div>
</template>

<script setup>
/**
 * EmptyState 空状态组件
 * @param {string} type - 类型：loading | error | empty
 * @param {Array} messages - 轮播文案列表
 * @param {number} interval - 轮播间隔（毫秒）
 */
import { ref, computed, onMounted, onUnmounted } from 'vue'

const props = defineProps({
  type: {
    type: String,
    default: 'empty',
    validator: (val) => ['loading', 'error', 'empty'].includes(val)
  },
  messages: {
    type: Array,
    default: () => ['暂无数据']
  },
  interval: {
    type: Number,
    default: 3000
  }
})

// 根据类型选择图片
const imageSrc = computed(() => {
  if (props.type === 'loading') return '/loading.png'
  if (props.type === 'error') return '/error.png'
  return '/empty.png'
})

const altText = computed(() => {
  if (props.type === 'loading') return '加载中'
  if (props.type === 'error') return '加载失败'
  return '暂无数据'
})

// 轮播文案
const currentIndex = ref(0)
const currentMessage = computed(() => props.messages[currentIndex.value])

let timer = null

onMounted(() => {
  if (props.messages.length > 1) {
    timer = setInterval(() => {
      currentIndex.value = (currentIndex.value + 1) % props.messages.length
    }, props.interval)
  }
})

onUnmounted(() => {
  if (timer) clearInterval(timer)
})
</script>
```
